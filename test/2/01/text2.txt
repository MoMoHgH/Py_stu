#!/usr/bin/env python

import dns.resolver
import os
import httplib



iplist=[]
appdomain = 'www.taobao.com'


def get_iplist(domain=''):
    try:
        A = dns.resolver.query(domain , 'A')
    except Exception,e:
        print 'dns resolver Error:'+str(e)
        return
    AA = A.response.answer
    del AA[0]
#   AA[CNAME,A]  CNAME : have no address
    for i in AA:
        for j in i.items:
            iplist.append(j.address)
    return True
def checkip(ip):
    checkurl=ip+':80'
    getconect = ''
    httplib.socket.setdefaulttimeout(60)     #timeout
    conn = httplib.HTTPConnection(checkurl)
    try:
        conn.request('GET','/', headers = {'Host': appdomain})
        r = conn.getresponse()
        getcontent = r.read(15)
    finally:
        if getcontent == "<!DOCTYPE html>":
            print ip+'[ok,IP is narmal]'
        else:
            print ip+'[Error]'
            execfile('/home/hgh/test/1/02/send_mail.py')

if __name__ == '__main__':
    if get_iplist(appdomain) and len(iplist)>0:
        for i in iplist:
            checkip(i)
    else:
        print 'dns resolver Error.'
